Import-Module WebAdministration 
New-WebAppPool -name "ResumeAppPool"  -force
$appPool = Get-Item -name "ResumeAppPool" 
$appPool.processModel.identityType = "NetworkService"
$appPool.enable32BitAppOnWin64 = 1
$appPool | Set-Item
md "c:\Inetpub\Resume"
$site = $site = new-WebSite -name "Resume" 
                            -PhysicalPath "c:\Inetpub\Resume" 
                            -HostHeader ""
							-Port 8080
                            -ApplicationPool "ResumeAppPool" 
                            -force
							
New-NetFirewallRule -DisplayName "Allow traffice for resume" -Direction Inbound -LocalPort 8080 -Protocol TCP -Action Allow
